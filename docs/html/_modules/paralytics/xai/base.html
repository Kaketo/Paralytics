[docs]    def fit_explain(self, X, y=None, **fit_params):
        """Fit to data, then explain properties."""
        if y is None:
            return self.fit(X, **fit_params).explain(X)
        else:
            return self.fit(X, y, **fit_params).explain(X, y)
